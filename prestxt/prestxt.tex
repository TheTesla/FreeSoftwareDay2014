\section{Vorteile!?}
\begin{frame}
\frametitle{Vorteile!?}
\begin{itemize}
\item Wartbarkeit/Anpassbarkeit
\item Testbarkeit -- Compilerunabhängig
\item Fehler können von unabhängigen Personen gefunden werden
\begin{itemize}
    \item Wer hat schon Bock darauf das Kauderwelsch von Narzisten zu überprüfen, die nicht in der Lage sind mit dieser "`Arbeit"' Geld zu verdienen?
\end{itemize}
\item größerer Druck auf die Entwickler, bei Fehlern -- Nö, ist doch kostenloser Ramsch
\item daher meist zeitnahe Behebung -- Firefox-Versionsnummer als Beweis
\item bei GPL: Zwang Änderungen/Erweiterungen zu veröffentlichen
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{noch mehr Vorteile!}
\begin{itemize}
    \item Top Doku -- weil jeder auf der Welt die Software kostenlos ausprobieren und penibel dokumentieren kann, auch den Quellcode -- natürlich übersichtlich in Doxygen
    \item Top Support -- begeisterte Nutzer und Entwickler wollen kein Geld verdienen -- sie wollen Ruhm und Ehre, indem sie alle durch selbstlose Unterstützung von den Vorteilen ihrer Software überzeugen
    \item 100 \% zuverlässig und fehlerfrei -- ausgiebige Test durch viele unabhängige Tester -- Testen ist bei kommerzieller Software extrem teuer, d. h. nicht wirtschaftlich möglich
    \item schnelle Reaktion auf den Markt -- bevor kapitalistische CS-Nazis auf den Markt drängen
    \item Riesige Auswahl -- Entwicklung ist ja schließlich kostenlos
    \item Es gibt keine Sicherheitslücken! Wer den Quellcode durchgelesen hätte, hätte erkennen müssen, dass die Software nie für diesen Zweck gedacht war.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{die wichtigsten Tatsachen!}
\begin{itemize}
    \item Linux stürzt nie ab!
    \item Linux hat keine Sicherheitslücken!
    \item Linux ist das stabilste Betriebssystem!
    \item Linux ist fehlerfrei!
    \item Unter Linux gibt es Prinzipbedingt niemals Viren.
    \item der GCC enthält keine Fehler, insbesondere keine, die die Sicherheit erzeugter Programme beeinträchtigen könnten!
    \item Ohne den Quellecode zu kennen ist es absolut nicht möglich Software hinschitlich Sicherheitslücken oder anderer Schwachstellen zu untersuchen!
    \item Nur OS Verschlüsselungssoftware taugt leider nichts, jeder kann ja sehen wie sie funktioniert.
    \item Die deutsche Automobilindustrie setzt auf frei Software!
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{kaum zu Glauben!}
\begin{itemize}
    \item Die Industrie strebt ein hohes Maß an Markttransparenz und Wettbewerb an. Sie entwickelt daher modulare freie Software, die von den Wettbewerbern problemlos wiederverwendet und angepasst werden kann! 
\end{itemize}
\end{frame}
